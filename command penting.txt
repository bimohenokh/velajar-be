## run django with uvicorn for asynchronous support (kalau gk pakai gk bakal bisa) ##
python -m uvicorn taschoolassistant.asgi:application --reload


## load all fixtures ##
# bash
python manage.py loaddata taschoolassistant/*/fixtures/*.json

# powershell
$fixtures = Get-ChildItem -Recurse -Filter *.json | Where-Object { $_.FullName -like "*\fixtures\*" } | ForEach-Object { $_.FullName }
python manage.py loaddata $fixtures

# cmd  # blm gw coba tapi
@echo off
setlocal enabledelayedexpansion
set FIXTURES=
for /R %%f in (*.json) do (
    echo %%f | findstr /R /C:"\\fixtures\\.*\.json" >nul
    if not errorlevel 1 (
        set FIXTURES=!FIXTURES! "%%f"
    )
)
python manage.py loaddata !FIXTURES!
